#+begin_src bash
uname -r
#+end_src
** Turn off monitor
#+begin_src bash
xset dpms force off
#+end_src
** Set cursor to left pointer
#+begin_src bash
xsetroot -cursor_name Left_ptr
#+end_src
** Get no. of packages installed (official repos and AUR)
#+begin_src bash
pacman -Qeq | wc -l
#+end_src
